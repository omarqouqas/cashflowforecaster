-- Give 6 months Pro to specific users
  -- Run this in Supabase SQL Editor

  WITH target_users AS (
    SELECT id, email FROM auth.users
    WHERE email IN (
      'steven.riera@me.com',
      'schroeder.crystal@gmail.com',
      'itsdanmanole@gmail.com',
      'satwantsinghbansi90@gmail.com',
      'jeremy.hillier@gmail.com',
      'stan.filippin@gmail.com',
      'o.bernie@gmail.com'
    )
  )
  INSERT INTO subscriptions (
    user_id,
    tier,
    status,
    current_period_start,
    current_period_end,
    cancel_at_period_end,
    created_at,
    updated_at
  )
  SELECT
    id,
    'pro',
    'active',
    NOW(),
    NOW() + INTERVAL '6 months',
    false,
    NOW(),
    NOW()
  FROM target_users
  ON CONFLICT (user_id)
  DO UPDATE SET
    tier = 'pro',
    status = 'active',
    current_period_start = NOW(),
    current_period_end = NOW() + INTERVAL '6 months',
    cancel_at_period_end = false,
    updated_at = NOW();
